# 顺序表

## SqList.h

```c++
#pragma once
#include<iostream>
#include<malloc.h>
#define MaxSize 50
using namespace std;
//顺序表的类型定义
typedef int ElemType;
typedef struct
{
	ElemType data[MaxSize];
	int length;
}SqList;

//建立顺序表
void CreateList(SqList*& L, ElemType a[], int n) {
	for (int i = 0; i < n; i++)
		L->data[i] = a[i];
	L->length = n;
}

//初始化顺序表
void InitList(SqList*& L) { //初始化SqList
	L = (SqList*)malloc(sizeof(SqList));
	L->length = 0;
}

//摧毁顺序表
void DestroyList(SqList*& L) //销毁SqList
{
	free(L);
}

//判断顺序表是否为空
bool ListEmpty(SqList* L) {
	return(L->length == 0);
}

//求顺序表的长度
int ListLength(SqList* L) {
	return L->length;
}

//显示顺序表
void DispList(SqList* L)
{
	for (int i = 0; i < L->length; i++) {
		cout << L->data[i] << " ";
	}
	cout << endl;
}

//获取顺序表L第i个元素，用e返回
bool GetElem(SqList* L, int i, ElemType& e) {
	if (i<1 || i>L->length)
		return false;
	e = L->data[i - 1];
}

//在顺序表L第i个位置的元素，插入数据元素e
bool ListInsert(SqList*& L, int i, ElemType e) {
	int j;
	if (i<1 || i>L->length + 1)
		return false;
	i--;
	for (j = L->length; j > i; j--)
		L->data[j] = L->data[j - 1];
	L->data[i] = e;
	L->length++;
	return true;
}

//在顺序表L第i个位置的元素，删除数据元素e
bool ListDelete(SqList*& L, int i, ElemType& e) {
	int j;
	if (i<1 || i>L->length)
		return false;
	i--;
	e = L->data[i];
	for (j = i; j < L->length - 1; j++)
		L->data[j] = L->data[j + 1];
	L->length--;
	return true;
}

```


## 案例一
main.cpp

```c++
#include"SqList.h"

int main(int argc, char** argv) {
	SqList* L1;
	int a[8] = {2,0,0,5,2,0,2,1};
	InitList(L1);//初始化顺序表
	CreateList(L1,a,sizeof(a)/sizeof(int));//创建顺序表

	cout << "创建后顺序表内的值为："; DispList(L1);
		
	ListInsert(L1, 3, 0);//在顺序表第三个位置插入0
	ListInsert(L1, 3, 2);//在顺序表第三个位置插入2
	ListInsert(L1, 3, 5);//在顺序表第三个位置插入5
	cout << "添加520后的值为："; DispList(L1);

	int b;
	ListDelete(L1, 3, b);//在顺序表第三个位置删除5
	//cout << "被删除的数是：" << b << endl;
	ListDelete(L1, 3, b);//在顺序表第三个位置删除2
	//cout << "被删除的数是：" << b << endl;
	ListDelete(L1, 3, b);//在顺序表第三个位置删除0
	//cout << "被删除的数是：" << b << endl;

	cout << "删除520后的值为："; DispList(L1);

	return 0;
}
```
